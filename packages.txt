# byebyte — runtime dependencies, hard vs optional (phanspeed/tjmax's
# packages.txt is the reference shape for this split — UNIFY.md Wave B).

# --- hard (packaging's deb Depends: python3 (>= 3.8), systemd, openssh-client) ---
python3         # the interpreter itself
systemd         # byebyted.service, the update/sweep timers, `systemctl` calls
openssh-client  # `ssh-keygen -Y verify`: sutra_update's release-signature
                # check against release-signing/allowed_signers. Vendored
                # sutra_update.py fail-closes without it once armed (by
                # construction) — but ByeByte arms at birth (Wave B), so a
                # fresh install missing this dependency would have a
                # fail-closed-because-missing-dependency update path from
                # day one, not a safe one. Hard, not optional.

# --- optional (soft; every one degrades to a stdlib-only fallback or a
# one-line notice, never an error) ---------------------------------------
btrfs-progs   # `btrfs` CLI: subvolume/snapshot/qgroup accounting on btrfs
              # mounts (V2.M2, read-only — no `quota enable`, no snapshot
              # verbs). Absent: why/blame fall back to the plain walk with
              # a "btrfs accounting unavailable" note instead of an error.
snapd         # `snap` CLI: backs the snap-old purge category (M3). Absent:
              # that category's detector just returns no candidates.
libnotify-bin # `notify-send` CLI: the `sweep` verb's toast when an armed
              # category actually reclaims something (V2.M4, UNIFY.md's
              # notification spec). Absent, or no active login session for
              # the owner: sweep still runs and ledgers, it just can't show
              # a desktop toast — never an error.
